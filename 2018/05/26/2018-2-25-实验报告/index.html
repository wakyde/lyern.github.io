对每门课程成绩的人数的统计

实验报告

编辑者:凌汪洪

实验名称
对某门课程统计某个分数段的人数

运行环境
win10 X64

编译环境
python3.6

编译软件
pycharm-5.0

依赖文件

代码实现

#! /usr/bin/env python
   #coding=utf-8


 import xlrd # excel数据包
 Perfect = 0 # 优秀的人数
 Good = 0    # 良好的人数
 Medium = 0  # 中等的人数
 Pass = 0    # 及格的人数
 Bad = 0     # 不及格的人数
 n = 0
 classNo ={"大学英语2":2,"离散数学":3,"高级语言程序设计":4,"面向对象程序设计":5,"数字逻辑":6,"数据结构":7} # 课程编号

 xls = xlrd.open_workbook('F:\\Python\\Homework\\grade.xlsx') # 打开成绩单
 sheet = xls.sheets()[0] # 确定成绩表
 print("请输入课程名:(大学英语2,离散数学,高级语言程序设计,面向对象程序设计,数字逻辑,数据结构)")
 className = input()  #输入课程名
 Number = classNo[className]
 class_grade = sheet.col_values(Number)

 for grade in range(1,34):
     n = n + 1
     grade = class_grade[n]
     if grade >=90:
         Perfect = Perfect + 1
     elif grade >= 80 and grade < 90:
         Good = Good + 1
     elif grade >= 70 and grade < 80:
         Medium = Medium +1
     elif grade >= 60 and grade < 70:
         Pass = Pass + 1
     elif grade < 60:
         Bad = Bad + 1

 print("优秀人数:%d" % Perfect)
 print("良好人数:%d" % Good)
 print("中等人数:%d" % Medium)
 print("及格人数:%d" % Pass)
 print("不及格人数:%d" % Bad)
